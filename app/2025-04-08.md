        );
      }
    }

      Person? person;
    
      SecondPage({@required this.person});
    
    
      @override
      Widget build(BuildContext context){
        final args = ModalRoute.of(context)!.settings.arguments as Map;
        person = args['person'];
    
        return Scaffold(
          appBar: AppBar(
            title:Text('Second'),
          ),
          body:
          Column(
            children: [
              Text('${person?.name}, ${person?.age}'),

                  Navigator.pop(
                    context, 'ok');
                },
                child: Text('이전 페이지로'),
              ),
            ],
          )
        );
      }
    }
